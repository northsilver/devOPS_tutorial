hosts=(192.168.0.1 173.194.222.113 87.250.250.242)
timeout=5
req=0
while (($req == 0))
do
    for h in ${hosts[@]}
    do
	curl -I -s --connect-timeout $timeout $h:80 >/dev/null
	req=$?
	if (($req != 0))
	then
	    echo "Error for =" $h status=$req >>hosts.log
	fi
    done
done
